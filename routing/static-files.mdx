title: Static Files

ZIO HTTP provides built-in support for serving static files from the filesystem or classpath resources. This is useful for serving assets like HTML, CSS, JavaScript, images, and downloadable files.

## Serving Files from Filesystem

Use `Handler.fromFile` or `Handler.fromFileZIO` to serve files from the filesystem:

```scala
import zio.http._
import java.io.File
import java.nio.charset.StandardCharsets

val fileRoute = Method.GET / "download" / "report.pdf" -> 
  Handler.fromFile(new File("/path/to/report.pdf"))

// With custom charset
val textFileRoute = Method.GET / "readme.txt" ->
  Handler.fromFile(new File("/path/to/readme.txt"), StandardCharsets.UTF_8)
```

## Serving Files with ZIO Effects

Use `Handler.fromFileZIO` when file access involves effects:

```scala
import zio.http._
import zio._
import java.io.File

val dynamicFileRoute = Method.GET / "files" / string("filename") -> 
  handler { (filename: String, req: Request) =>
    val getFile = ZIO.attempt {
      new File(s"/var/data/$filename")
    }
    
    Handler.fromFileZIO(getFile).apply(req)
  }
```

## Serving Resources from Classpath

Serve resources bundled with your application:

```scala
import zio.http._

val resourceRoute = Method.GET / "assets" / "logo.png" ->
  Handler.fromResource("static/images/logo.png")
```

## Static File Server with Middleware

Use middleware to serve an entire directory:

```scala
import zio.http._

val app = Routes(
  Method.GET / "api" / "health" -> handler(Response.ok),
  Method.GET / "api" / "users" -> handler(Response.text("Users")),
).toHttpApp

// Serve static files from /public directory at /static/*
val withStatic = app @@ Middleware.serveDirectory(
  Path("/static"),
  new java.io.File("/var/www/public")
)

// Now GET /static/styles/main.css serves /var/www/public/styles/main.css
```

## Serving Static Resources

Serve classpath resources with middleware:

```scala
import zio.http._

val app = Routes(
  Method.GET / "api" / "data" -> handler(Response.text("Data"))
).toHttpApp

// Serve resources from classpath at /assets/*
val withAssets = app @@ Middleware.serveResources(
  Path("/assets"),
  "public" // resources/public/
)

// Now GET /assets/css/style.css serves resources/public/css/style.css
```

## Dynamic File Paths

Serve files based on path parameters:

```scala
import zio.http._
import zio.http.codec.PathCodec._
import java.io.File

val fileServerRoute = Method.GET / "files" / trailing -> 
  handler { (path: Path, req: Request) =>
    val filePath = s"/var/data${path.encode}"
    val file = new File(filePath)
    
    Handler.fromFileZIO(ZIO.succeed(file)).apply(req)
  }

// GET /files/documents/2024/report.pdf
// Serves: /var/data/documents/2024/report.pdf
```

## Content Type Detection

ZIO HTTP automatically detects content types based on file extensions:

```scala
import zio.http._
import java.io.File

val imageRoute = Method.GET / "images" / string("name") ->
  handler { (name: String, req: Request) =>
    // Content-Type is automatically set based on file extension
    Handler.fromFile(new File(s"/images/$name")).apply(req)
  }

// GET /images/photo.jpg -> Content-Type: image/jpeg
// GET /images/logo.png -> Content-Type: image/png
// GET /images/icon.svg -> Content-Type: image/svg+xml
```

## Custom Media Types

Override the detected media type:

```scala
import zio.http._
import java.io.File

val route = Method.GET / "data.json" -> handler { (req: Request) =>
  Handler.fromFile(new File("/data/export.json"))
    .apply(req)
    .map(_.addHeader(Header.ContentType(MediaType.application.json)))
}
```

## Range Requests (Partial Content)

ZIO HTTP automatically handles HTTP range requests for efficient large file transfers:

```scala
import zio.http._
import java.io.File

val videoRoute = Method.GET / "videos" / string("name") ->
  handler { (name: String, req: Request) =>
    val videoFile = new File(s"/media/videos/$name")
    // Automatically handles Range header for video streaming
    Handler.fromFileZIO(ZIO.succeed(videoFile)).apply(req)
  }

// Client can request byte ranges:
// GET /videos/movie.mp4
// Range: bytes=0-1023
// 
// Response includes:
// Status: 206 Partial Content
// Content-Range: bytes 0-1023/1048576
```

## File Download with Custom Name

Force download with a specific filename:

```scala
import zio.http._
import java.io.File

val downloadRoute = Method.GET / "download" / int("reportId") ->
  handler { (reportId: Int, req: Request) =>
    val file = new File(s"/reports/report_$reportId.pdf")
    
    Handler.fromFileZIO(ZIO.succeed(file))
      .apply(req)
      .map { response =>
        response.addHeader(
          Header.ContentDisposition.attachment(s"monthly_report_$reportId.pdf")
        )
      }
  }

// Browser will download as "monthly_report_123.pdf"
```

## Conditional File Serving

Serve files with caching and conditional requests:

```scala
import zio.http._
import java.io.File
import java.time.Instant

val cachedFileRoute = Method.GET / "assets" / string("file") ->
  handler { (file: String, req: Request) =>
    val target = new File(s"/static/$file")
    
    Handler.fromFileZIO(ZIO.succeed(target))
      .apply(req)
      .map { response =>
        val lastModified = Instant.ofEpochMilli(target.lastModified())
        
        response
          .addHeader(Header.LastModified(
            java.time.ZonedDateTime.ofInstant(lastModified, java.time.ZoneOffset.UTC)
          ))
          .addHeader(Header.CacheControl.MaxAge(java.time.Duration.ofDays(7)))
      }
  }
```

## Error Handling

Handle file not found and access errors:

```scala
import zio.http._
import java.io.{File, FileNotFoundException}
import java.nio.file.AccessDeniedException

val safeFileRoute = Method.GET / "files" / string("name") ->
  handler { (name: String, req: Request) =>
    val file = new File(s"/data/$name")
    
    Handler.fromFileZIO(ZIO.succeed(file))
      .apply(req)
      .catchAll {
        case _: FileNotFoundException => 
          ZIO.succeed(Response.notFound(s"File '$name' not found"))
        case _: AccessDeniedException => 
          ZIO.succeed(Response.forbidden("Access denied"))
        case e: Throwable => 
          ZIO.succeed(Response.internalServerError(s"Error: ${e.getMessage}"))
      }
  }
```

## Security Considerations

### Path Traversal Prevention

Protect against directory traversal attacks:

```scala
import zio.http._
import java.io.File
import java.nio.file.{Path => NioPath, Paths}

def isPathSafe(baseDir: NioPath, requestedPath: String): Boolean = {
  val normalized = baseDir.resolve(requestedPath).normalize()
  normalized.startsWith(baseDir)
}

val secureFileRoute = Method.GET / "files" / trailing ->
  handler { (path: Path, req: Request) =>
    val baseDir = Paths.get("/var/data")
    val requestedPath = path.encode
    
    if (isPathSafe(baseDir, requestedPath)) {
      val file = baseDir.resolve(requestedPath).toFile
      Handler.fromFileZIO(ZIO.succeed(file)).apply(req)
    } else {
      ZIO.succeed(Response.forbidden("Invalid path"))
    }
  }
```

### File Type Restrictions

Restrict served file types:

```scala
import zio.http._
import java.io.File

val allowedExtensions = Set(".jpg", ".png", ".gif", ".pdf", ".txt")

val restrictedFileRoute = Method.GET / "downloads" / string("filename") ->
  handler { (filename: String, req: Request) =>
    val extension = filename.substring(filename.lastIndexOf('.'))
    
    if (allowedExtensions.contains(extension.toLowerCase)) {
      val file = new File(s"/downloads/$filename")
      Handler.fromFileZIO(ZIO.succeed(file)).apply(req)
    } else {
      ZIO.succeed(Response.forbidden("File type not allowed"))
    }
  }
```

## SPA (Single Page Application) Support

Serve a single-page application with fallback to index.html:

```scala
import zio.http._
import java.io.File

def serveSPA(publicDir: File): Routes[Any, Response] = {
  val indexFile = new File(publicDir, "index.html")
  
  Routes(
    // Try to serve static file
    Method.GET / trailing -> handler { (path: Path, req: Request) =>
      val file = new File(publicDir, path.encode)
      
      if (file.exists() && file.isFile) {
        Handler.fromFileZIO(ZIO.succeed(file)).apply(req)
      } else {
        // Fallback to index.html for client-side routing
        Handler.fromFileZIO(ZIO.succeed(indexFile)).apply(req)
      }
    }
  )
}

val spaRoutes = serveSPA(new File("/var/www/spa/build"))
```

## Streaming Large Files

Large files are automatically streamed efficiently:

```scala
import zio.http._
import java.io.File

val largeFileRoute = Method.GET / "downloads" / "large-dataset.csv" ->
  Handler.fromFile(
    new File("/data/large-dataset.csv")
    // File is streamed in chunks, not loaded entirely into memory
  )
```

## Multiple Static Directories

Serve multiple static directories:

```scala
import zio.http._
import java.io.File

val app = Routes(
  Method.GET / "api" / "status" -> handler(Response.ok)
).toHttpApp

val withStaticFiles = app @@
  Middleware.serveDirectory(Path("/css"), new File("/var/www/css")) @@
  Middleware.serveDirectory(Path("/js"), new File("/var/www/js")) @@
  Middleware.serveDirectory(Path("/images"), new File("/var/www/images"))
```

## Common Patterns

### Static Asset Server

```scala
import zio.http._
import java.io.File

val staticAssets = Routes(
  Method.GET / "assets" / trailing -> handler { (path: Path, req: Request) =>
    val file = new File(s"/var/www/assets${path.encode}")
    
    Handler.fromFileZIO(ZIO.succeed(file))
      .apply(req)
      .map(_.addHeader(Header.CacheControl.MaxAge(java.time.Duration.ofDays(30))))
  }
)
```

### File Upload and Download

```scala
import zio.http._
import java.io.File
import java.nio.file.{Files, Paths}

val fileRoutes = Routes(
  // Upload
  Method.POST / "upload" -> handler { (req: Request) =>
    for {
      body <- req.body.asMultipartForm
      file <- body.get("file").map(_.asFile).getOrElse(ZIO.fail(new Exception("No file")))
      _ <- ZIO.attemptBlocking {
        Files.copy(file.toPath, Paths.get(s"/uploads/${file.getName}"))
      }
    } yield Response.text("File uploaded")
  },
  
  // Download
  Method.GET / "download" / string("filename") -> handler { (filename: String, req: Request) =>
    Handler.fromFile(new File(s"/uploads/$filename")).apply(req)
  }
)
```

### Dynamic Documentation Server

```scala
import zio.http._
import java.io.File

val docsServer = Routes(
  Method.GET / "docs" / trailing -> handler { (path: Path, req: Request) =>
    val docPath = if (path.isEmpty) "index.html" else path.encode
    val file = new File(s"/docs/build/$docPath")
    
    Handler.fromFileZIO(ZIO.succeed(file))
      .apply(req)
      .map(_.addHeader(Header.ContentType(MediaType.text.html)))
      .catchAll(_ => 
        ZIO.succeed(Response.notFound("Documentation page not found"))
      )
  }
)
```
