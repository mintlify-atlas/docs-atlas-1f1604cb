ZIO HTTP provides a powerful WebSocket client that allows you to establish WebSocket connections, send and receive messages, and handle connection lifecycle events.

## Basic WebSocket Connection

### Using WebSocketApp

The core of WebSocket client functionality is `WebSocketApp`:

```scala
import zio._
import zio.http._
import zio.http.ChannelEvent._

val socketApp: WebSocketApp[Any] = 
  Handler.webSocket { channel =>
    channel.receiveAll {
      // Handle connection established
      case UserEventTriggered(UserEvent.HandshakeComplete) =>
        channel.send(Read(WebSocketFrame.text("Hello, Server!")))
      
      // Handle incoming text messages
      case Read(WebSocketFrame.Text(text)) =>
        Console.printLine(s"Received: $text").ignore
      
      // Handle connection close
      case Read(WebSocketFrame.Close(status, reason)) =>
        Console.printLine(s"Connection closed: $status - $reason").ignore
      
      case _ =>
        ZIO.unit
    }
  }
```

## Handling WebSocket Events

### Channel Events

WebSocket channels emit various events:

```scala
import zio.http.ChannelEvent._

val socketApp = Handler.webSocket { channel =>
  channel.receiveAll {
    // Connection established
    case UserEventTriggered(UserEvent.HandshakeComplete) =>
      Console.printLine("WebSocket connected!") *>
      channel.send(Read(WebSocketFrame.text("Hello")))
    
    // Received text frame
    case Read(WebSocketFrame.Text(text)) =>
      Console.printLine(s"Text: $text")
    
    // Received binary frame
    case Read(WebSocketFrame.Binary(data)) =>
      Console.printLine(s"Binary: ${data.size} bytes")
    
    // Received ping
    case Read(WebSocketFrame.Ping) =>
      channel.send(Read(WebSocketFrame.Pong))
    
    // Received pong
    case Read(WebSocketFrame.Pong) =>
      Console.printLine("Pong received")
    
    // Connection closing
    case Read(WebSocketFrame.Close(status, reason)) =>
      Console.printLine(s"Close: $status - $reason")
    
    // Catch all
    case _ =>
      ZIO.unit
  }
}
```

## Complete WebSocket Example

```scala
import zio._
import zio.http._
import zio.http.ChannelEvent.{Read, UserEvent, UserEventTriggered}

object WebSocketSimpleClient extends ZIOAppDefault {

  val url = "ws://ws.vi-server.org/mirror"

  val socketApp: WebSocketApp[Any] =
    Handler.webSocket { channel =>
      channel.receiveAll {
        // Send a "foo" message to the server once the connection is established
        case UserEventTriggered(UserEvent.HandshakeComplete) =>
          channel.send(Read(WebSocketFrame.text("foo")))

        // Send a "bar" if the server sends a "foo"
        case Read(WebSocketFrame.Text("foo")) =>
          channel.send(Read(WebSocketFrame.text("bar")))

        // Close the connection if the server sends a "bar"
        case Read(WebSocketFrame.Text("bar")) =>
          ZIO.succeed(println("Goodbye!")) *> 
          channel.send(Read(WebSocketFrame.close(1000)))

        case _ =>
          ZIO.unit
      }
    }

  val app: ZIO[Client, Throwable, Response] =
    ZIO.scoped(socketApp.connect(url) *> ZIO.never)

  val run: ZIO[Any, Throwable, Any] =
    app.provide(Client.default)
}
```

## Environment and Dependencies

### Providing Environment

```scala
// WebSocketApp with dependencies
val socketAppWithEnv: WebSocketApp[MyService] = 
  Handler.webSocket { channel =>
    channel.receiveAll {
      case Read(WebSocketFrame.Text(text)) =>
        for {
          service <- ZIO.service[MyService]
          result <- service.processMessage(text)
          _ <- channel.send(Read(WebSocketFrame.text(result)))
        } yield ()
      
      case _ => ZIO.unit
    }
  }

// Provide environment
val providedApp: WebSocketApp[Any] = 
  socketAppWithEnv.provideEnvironment(

val layeredApp: WebSocketApp[Any] =
  socketAppWithEnv.provideLayer(MyService.live)
```

## Advanced Example: Message Echo with State

```scala
import zio._
import zio.http._
import zio.http.ChannelEvent._

object StatefulWebSocketClient extends ZIOAppDefault {
  
  val socketApp = Handler.webSocket { channel =>
    for {
      // Create message counter
      counter <- Ref.make(0)
      
      // Handle messages
      _ <- channel.receiveAll {
        case UserEventTriggered(UserEvent.HandshakeComplete) =>
          Console.printLine("Connected to WebSocket server") *>
          channel.send(Read(WebSocketFrame.text("Hello")))
        
        case Read(WebSocketFrame.Text(text)) =>
          for {
            count <- counter.updateAndGet(_ + 1)
            _ <- Console.printLine(s"Message #$count: $text")
            response = s"Echo #$count: $text"
            _ <- channel.send(Read(WebSocketFrame.text(response)))
            // Close after 5 messages
            _ <- channel.send(Read(WebSocketFrame.close(1000))).when(count >= 5)
          } yield ()
        
        case Read(WebSocketFrame.Close(status, reason)) =>
          Console.printLine(s"Connection closed: $status - $reason")
        
        case _ => ZIO.unit
      }
    } yield ()
  }
  
  val app = ZIO.scoped {
    socketApp.connect("wss://echo.websocket.org") *> ZIO.never
  }
  
  override val run = app.provide(Client.default)
}
```

## WebSocket Client with SSL

```scala
import zio._
import zio.http._
import zio.http.netty.NettyConfig

object SecureWebSocketClient extends ZIOAppDefault {
  
  val sslConfig = ClientSSLConfig.FromTrustStoreResource(
    trustStorePath = "truststore.jks",
    trustStorePassword = "changeit"
  )
  
  val clientConfig = ZClient.Config.default.ssl(sslConfig)
  
  val socketApp: WebSocketApp[Any] = 
    Handler.webSocket { channel =>
      channel.receiveAll {
        case UserEventTriggered(UserEvent.HandshakeComplete) =>
          channel.send(Read(WebSocketFrame.text("Secure hello")))
        
        case Read(WebSocketFrame.Text(text)) =>
          Console.printLine(s"Secure message: $text")
        
        case _ => ZIO.unit
      }
    }
  
  val app = ZIO.scoped {
    socketApp.connect("wss://secure.example.com/ws") *> ZIO.never
  }
  
  override val run = app.provide(
    ZLayer.succeed(clientConfig),
    Client.live,
    ZLayer.succeed(NettyConfig.default),
    DnsResolver.default,
  )
}
```

## Best Practices

### Resource Management

Always use `ZIO.scoped` to ensure WebSocket connections are properly closed:

```scala
// Good: Properly scoped
val app = ZIO.scoped {
  socketApp.connect(url) *> longRunningLogic
}

// Bad: Missing scope - connection may leak
val app = socketApp.connect(url) // Compile error - missing Scope
```

### Error Handling

Handle connection errors gracefully:

```scala
val resilientApp = socketApp
  .tapErrorZIO { error =>
    Console.printLine(s"Connection error: ${error.getMessage}") *>
    // Log or report error
    ZIO.unit
  }
```

### Graceful Shutdown

Close connections with proper status codes:

```scala
// Normal closure
channel.send(Read(WebSocketFrame.close(1000, "Normal closure")))

// Going away
channel.send(Read(WebSocketFrame.close(1001, "Client going away")))

// Protocol error
channel.send(Read(WebSocketFrame.close(1002, "Protocol error")))
```

### Keep-Alive

Implement ping/pong for connection health:

```scala
val keepAliveApp = Handler.webSocket { channel =>
  for {
    // Send ping every 30 seconds
    _ <- (channel.send(Read(WebSocketFrame.Ping)) *> 
          ZIO.sleep(30.seconds)).forever.fork
    
    // Handle messages
    _ <- channel.receiveAll {
      case Read(WebSocketFrame.Pong) =>
        // Connection is alive
        ZIO.unit
      
      case other => handleMessage(other)
    }
  } yield ()
}
```
